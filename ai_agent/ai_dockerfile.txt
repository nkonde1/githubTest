# AI Agent Dockerfile for Embedded Finance Platform

FROM python:3.11-slim

# Set environment variables
ENV PYTHONUNBUFFERED=1
ENV PYTHONDONTWRITEBYTECODE=1
ENV PYTHONPATH="/app"

# Create non-root user for security
RUN groupadd -r aiagent && useradd -r -g aiagent aiagent

# Install system dependencies
RUN apt-get update && apt-get install -y \
    gcc \
    g++ \
    curl \
    && rm -rf /var/lib/apt/lists/*

# Set working directory
WORKDIR /app

# Copy requirements first for better layer caching
COPY requirements.txt .

# Install Python dependencies
RUN pip install --no-cache-dir --upgrade pip && \
    pip install --no-cache-dir -r requirements.txt

# Copy application code
COPY llama_service/ ./llama_service/
COPY *.py ./

# Create necessary directories
RUN mkdir -p /app/logs /app/data /app/prompts

# Copy prompt files
COPY llama_service/prompts/ ./llama_service/prompts/

# Set proper permissions
RUN chown -R aiagent:aiagent /app
USER aiagent

# Health check
HEALTHCHECK --interval=30s --timeout=10s --start-period=5s --retries=3 \
    CMD curl -f http://localhost:8080/health || exit 1

# Expose port
EXPOSE 8080

# Default command
CMD ["python", "-m", "llama_service.main"]
